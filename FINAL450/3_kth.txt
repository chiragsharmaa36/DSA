// Find the "Kth" max and min element of an array
// Expected Time Complexity: O(n)
// Expected Auxiliary Space: O(log(n))
#include <iostream>
#include <climits>

using namespace std;

// k is the kth smallest number

int kth(int arr[], int l, int r, int k)
{
    int i;
    int m = arr[l];
    int ar[r + 1];
    int serial;

    for (int j = 0; j <= k; j++)
    {
        for (i = l; i <= r; i++)
        {
            if (m > arr[i])
            {
                m = arr[i];
                serial = i;
            }
        }
        ar[j] = m;
        m=INT_MAX;
        arr[serial] = 200;
    }
    return ar[k - 1];
}

int main()
{
    int n = 6, k;
    int arr[n] = {7, 10, 4,3, 20, 15};
    cout << "k = ";
    cin >> k;
    cout << "ans = " << kth(arr, 0, n - 1, k);

    return 0;
}
